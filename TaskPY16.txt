N = int(input("Введите количество элементов в массиве: "))
A = list(map(int, input("Введите элементы массива через пробел: ").split()))
X = int(input("Введите искомое число: "))

count = 0

for i in A:
    if i == X:
        count += 1

print("Число", X, "встречается в массиве", A, count, "раз(а).")
